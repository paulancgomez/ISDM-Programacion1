function definirTamanio(&$N){
    $N= Rand(3,6); 
}

//CARGA MATRIZ

function cargaMatriz(&$matriz, $n){
    for($i=0; $i<$n; $i++){
        for($j=0; $j<$n; $j++){
            $matriz[$i][$j] = Rand(10,99);
        }
    }
}

--------------------------------------------------------------------------------------
//MUESTRA MATRIZ

function muestraMatriz($matriz, $n){
    for($i=0; $i<$n; $i++){
        echo "<br>";
        for($j=0; $j<$n; $j++){
            echo $matriz[$i][$j] . " || ";
        }
    }
    echo "<br>";
}

---------------------------------------------------------------------------------------
//SUMA MATRICES

function sumaMatriz($A, $B, &$C, $n){
    for($i=0; $i<$n; $i++){
        for($j=0; $j<$n; $j++){
            $C[$i][$j] = $A[$i][$j] + $B[$i][$j];
        }
    }
}
