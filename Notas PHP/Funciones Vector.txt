//MENOR

function menor($vector){
    $menor= $vector[0];
    for($i=0; $i<count($vector); $i++){
        if($vector[$i] < $menor){
            $menor= $vector[$i];
        }
    }
    return $menor;
}

function menorSort($vector){
    sort($vector);
    return $vector[0];
}

-----------------------------------------------------------------------------------------
//MAYOR

function mayor($vector){
    $mayor= $vector[0];
    for($i=0; $i<count($vector); $i++){
        if($vector[$i] > $mayor){
            $mayor= $vector[$i];
        }
    }
    return $mayor;
}

function mayorRsort($vector){
    rsort($vector);
    return $vector[0];
}

//PROMEDIO

function calcular_promedio($vector){
        for($i=0; $i<count($vector); $i++){
            $suma= $suma + $vector[$i];
        }
        $promedio= $suma / count($vector);
        return $promedio;
    }
